# Distillation recipe deps. Pinned loosely — these libraries move
# fast and pinning to patch versions ages badly. Tested with the
# minor versions shown.
#
# torch is intentionally left to the user to install — the right
# wheel depends on the platform (CUDA / ROCm / MPS / CPU). Install
# it first, e.g.:
#   pip install torch --index-url https://download.pytorch.org/whl/cu121
# then:
#   pip install -r requirements.txt

# Core training stack.
transformers>=4.45,<5.0
trl>=0.11,<1.0
peft>=0.13,<1.0
datasets>=3.0,<4.0

# 4-bit quantization for the frozen base (QLoRA). CUDA-only — on
# Apple Silicon set --no-4bit and the train script will skip
# bitsandbytes and fall back to bf16/fp16.
bitsandbytes>=0.44,<1.0; platform_system != "Darwin"

# Tokenizer dependencies. The transformers install pulls these in,
# but pinning explicitly avoids surprises when transformers updates.
tokenizers>=0.20,<1.0
sentencepiece>=0.2.0
protobuf>=4.25

# Misc.
# accelerate 1.13+ produces NaN losses from step 1 when training in
# bf16 on Apple Silicon (MPS) — verified 2026-07-02 with torch 2.12.x,
# transformers 5.12, trl 1.7: loss explodes to e+05, grad_norm nan.
# Keep below 1.13 until upstream fixes MPS mixed-precision handling.
accelerate>=1.0,<1.13
safetensors>=0.4
huggingface-hub>=0.26

# GGUF export (export_gguf.py): llama.cpp's convert_hf_to_gguf.py imports
# this to write the GGUF. Required for the `distill export` step.
gguf>=0.10

# Optional: required only when using a Claude model as the teacher
# (--model claude-opus-4-7, --model claude-sonnet-4-6, etc.)
anthropic>=0.40
